void setup() {
  Serial.begin(9600);
}

void loop() {
  int   sensorValue = analogRead(A0);
  float voltage     = sensorValue * (5.0 / 1023.0); // Calculate the actual voltage at sensor out

  Serial.print(sensorValue);
  Serial.print(",");
  Serial.println(voltage/0.1);
  delay(100);        
}
